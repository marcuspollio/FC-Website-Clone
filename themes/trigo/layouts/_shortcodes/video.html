{{- /* SPDX-License-Identifier: MIT SPDX-FileCopyrightText: 2026 FreeCAD SPDX-FileNotice: Part of the Trigo theme for Hugo. */ -}} {{- /* Video shortcode: @param {string} src – Path to video resource (page bundle). @param {string} [poster] – Optional poster image resource. @return {HTML} Render video with optional poster and fallback text if video missing. @example {{< video src="video.mp4" poster="cover.jpg" >}} @example {{< video "video.mp4" "cover.jpg" >}} */ -}} {{- $src := cond (.IsNamedParams) (.Get "src") (.Get 0) -}} {{- if $src -}} {{- $poster := cond (.IsNamedParams) (.Get "poster") (.Get 1) | default "" -}} {{- $url := urls.Parse $src -}} {{- $is_remote := $url.IsAbs -}} {{- $video := "" -}} {{- $poster_res := "" -}} {{- if $is_remote -}} {{- $video = $src -}} {{- else -}} {{- with .Page.Resources.GetMatch $src -}} {{- $video = .RelPermalink -}} {{- end -}} {{- end -}} {{- if $poster -}} {{- if (urls.Parse $poster).IsAbs -}} {{- with try (resources.GetRemote $poster) -}} {{- with .Value }} {{- $poster_res = .RelPermalink -}} {{- else }} {{- $poster_res = $poster -}} {{- end -}} {{- end -}} {{- else -}} {{- with .Page.Resources.GetMatch $poster -}} {{- $poster_res = .RelPermalink -}} {{- end -}} {{- end -}} {{- end -}} {{- if and (not $poster_res) (not $is_remote) -}} {{- $filename := replace $src (printf ".%s" (path.Ext $src)) "" -}} {{- range $ext := slice "jpg" "jpeg" "png" "webp" -}} {{- with (first 1 ($.Page.Resources.Match (printf "%s*.%s" $filename $ext))) -}} {{- $poster_res = (index . 0).RelPermalink -}} {{- break -}} {{- end -}} {{- end -}} {{- end -}} {{- $video_fallback := i18n "video_fallback" | default "Your browser does not support the video tag." -}} {{- else -}} {{- warnf "%q shortcode: missing required 'src' argument on %s" .Name .Position -}} {{- end -}}
