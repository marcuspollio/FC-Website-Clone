{{- /* SPDX-License-Identifier: MIT SPDX-FileCopyrightText: 2026 FreeCAD SPDX-FileNotice: Part of the Trigo theme for Hugo. */ -}} {{- /* Process image resource: - Build responsive variants for processable images. - Support non-processable image formats with metadata. @returns {template.HTML} */ -}} {{- $image := .image -}} {{- $src := $image.RelPermalink -}} {{- $alt := .alt | default (i18n "image_fallback" | default "Image fallback") -}} {{- $attributes := .attributes | default dict -}} {{- $title := .title -}} {{- $resize_format := site.Params.resize_format | default "webp" -}} {{- $resize_dimensions := site.Params.resize_dimensions | default (slice 400 800 1600) -}} {{- $loading := site.Params.image_loading | default "lazy" -}} {{- $priority := site.Params.image_priority | default "low" -}} {{- $decoding := site.Params.image_decoding | default "async" -}} {{- $width := 0 -}} {{- $height := 0 -}} {{- $srcset := slice -}} {{- $sizes := slice -}} {{- /* Responsive and processable image (e.g. bmp, gif, jpg, png, tif, webp). */ -}} {{- if reflect.IsImageResourceProcessable $image -}} {{- $width = $image.Width -}} {{- $height = $image.Height -}} {{- range $resize_dimensions -}} {{- $resize_width := int . -}} {{- if lt $resize_width $width -}} {{- $resized := $image.Resize (printf "%dx %s" $resize_width $resize_format) -}} {{- $srcset = $srcset | append (printf "%s %dw" $resized.RelPermalink $resize_width) -}} {{- $sizes = $sizes | append (printf "(max-width: %dpx) %dpx" $resize_width $resize_width) -}} {{- end -}} {{- end -}} {{- $srcset = $srcset | append (printf "%s %dw" $src $width) -}} {{- $sizes = $sizes | append "95vw" -}} {{- /* Non-processable and non-resizable image with metadata (e.g. avif, heic, heif). */ -}} {{- else if reflect.IsImageResourceWithMeta $image -}} {{- $width = $image.Width -}} {{- $height = $image.Height -}} {{- end -}}
