{{- /* SPDX-License-Identifier: MIT SPDX-FileCopyrightText: 2026 FreeCAD SPDX-FileNotice: Part of the Trigo theme for Hugo. */ -}} {{- /* Pagination for Single pages: - Set Navigation links to Previous and Next pages within current Section - Set links to Previous and Next Sections if first and end pages in Section. @returns {template.HTML} */ -}} {{- $page := . -}} {{- $current_section := .CurrentSection -}} {{- $pages := $current_section.RegularPages | default $page.Parent.Sections -}} {{- with $pages -}}
{{- range $direction, $icon := dict "prev" "tri_right" "next" "tri_left" -}} {{- $item := "" -}} {{- $type := "" -}} {{- if eq $direction "prev" -}} {{- $item = $pages.Prev $page -}} {{- else -}} {{- $item = $pages.Next $page -}} {{- end -}} {{- if and $item $page.IsPage -}} {{- $type = "article" -}} {{- else if not $current_section.Parent.IsHome -}} {{- $siblings := $current_section.Parent.Sections.ByWeight -}} {{- if eq $direction "prev" -}} {{- $item = $siblings.Prev $current_section -}} {{- else -}} {{- $item = $siblings.Next $current_section -}} {{- end -}} {{- if $item -}} {{- $type = "section" -}} {{- $icon = "folder_closed" -}} {{- end -}} {{- end -}} {{- with $item -}} {{- $item_page := .Page -}} {{- $href := .RelPermalink -}} {{- with $item_page.Params.redirect -}} {{- if and $item_page.IsSection (eq . "auto") -}} {{- $item_pages := sort $item_page.Pages -}} {{- with index $item_pages 0 -}} {{- $href = .RelPermalink -}} {{- end -}} {{- else -}} {{- $target := site.GetPage . -}} {{- with $target -}} {{- $href = .RelPermalink -}} {{- else -}} {{- $href = . -}} {{- end -}} {{- end -}} {{- end -}} {{- $is_external := and (hasPrefix $href "http") (not (hasPrefix $href site.BaseURL)) -}} {{- if eq $direction "next" -}} {{- partialCached "utils/icon.html" $icon $icon -}} {{- end -}} {{- title .LinkTitle -}} {{- if $is_external -}} {{- partialCached "utils/icon.html" "external" "external" -}} {{- end -}} {{- if eq $direction "prev" -}} {{- partialCached "utils/icon.html" $icon $icon -}} {{- end -}} {{- if not hugo.IsProduction -}} {{- with .Weight -}} Weight: {{ . }} {{- end -}} {{- end -}} {{- end -}} {{- end -}}
{{- end -}}
