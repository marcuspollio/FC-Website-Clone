{{- /* SPDX-License-Identifier: MIT SPDX-FileCopyrightText: 2026 FreeCAD SPDX-FileNotice: Part of the Trigo theme for Hugo. */ -}} {{- /* Welcome contributors custom partial: - Render single-choice tree buttons from welcome front matter labels and link. @returns {template.HTML} */ -}} {{- $welcome := .Params.welcome -}} {{- with $welcome -}}
{{- range $i, $item := . -}} {{- if $item.children -}} {{- $id := printf "l1-%s" ($item.id | default (printf "%d" $i)) -}}
{{- partial "inline/welcome-level.html" (dict "nodes" $item.children "parent" $id "level" 2 ) -}}
{{- else if not .children -}} {{- $href := $item.url -}} {{- with $href -}} {{- with site.GetPage . -}} {{- $key := partialCached "utils/cache_key.html" . . -}} {{- $href = partialCached "utils/link_redirect.html" . $key -}} {{- end -}} {{- end -}} {{- $is_external := partialCached "utils/is_external.html" $href $href -}}
{{- $item.label -}} {{- if $is_external -}} {{- partialCached "utils/icon.html" "external" "external" -}} {{- end -}}
{{- end -}} {{- end -}}
{{- end -}} {{- define "_partials/inline/welcome-level.html" -}} {{- $nodes := .nodes -}} {{- $parent := .parent -}} {{- $level := .level -}}
{{- range $i, $node := $nodes -}} {{- $id := printf "%s-%s" $parent ($node.id | default (printf "%d" $i)) -}} {{- if $node.children -}}
{{- partial "inline/welcome-level.html" (dict "nodes" $node.children "parent" $id "level" (add $level 1) ) -}}
{{- else -}} {{- $href := $node.url -}} {{- with $href -}} {{- with site.GetPage . -}} {{- $key := partialCached "utils/cache_key.html" . . -}} {{- $href = partialCached "utils/link_redirect.html" . $key -}} {{- end -}} {{- end -}} {{- $is_external := partialCached "utils/is_external.html" $href $href -}} {{- $node.label -}} {{- if $is_external -}} {{- partialCached "utils/icon.html" "external" "external" -}} {{- end -}} {{- end -}} {{- end -}}
{{- end -}}
